t = []

  for filename in filenames:
    try:
      print(filename)
      f.append(filename)
      img = Image.open(input_dir+ filename)
      text = pytesseract.image_to_string(img, lang='jpn')
      t.append(text)
      print(text)
      print('-='*20)
    except:
      continue

outputfile = open(input_dir + "result.txt", "w")
for element in t:
  outputfile.write(element + "\n")
outputfile.close()
