```javascript
const cat = new Cat();
const settings = QuickSettings.create(window.innerWidth - 200);

// input:text
settings.addTest("Name", (v) => {
  cat.setName(v);
});
// checkbox
settings.addBoolean("Sleep", cat.isSleep, (v) => {
  cat.setSleep(v);
});
// button
settings.addButton("meow", () => {
  cat.meow();
});
```
